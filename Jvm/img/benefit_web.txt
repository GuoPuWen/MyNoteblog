    function request(method, url, param) {
      // var protocol = window.location.protocol.substring(0, window.location.protocol.length - 1)
      var independent = "{$independent}";
      if (independent == 1) {
        var h = window.location.host;
        var domain = h.replace('www', window.location.protocol + '//api');
      } else {
        var domain = 'https://api.haodanku.com';
      }


      var requestConfig = {}
      if (method === 'get') {
        param = param ? param : {}
        requestConfig = { method: method, url: domain + url, params: param }
      } else if (method === 'post') {
        param = param ? Qs.stringify(param, { arrayFormat: 'repeat' }) : {}
        requestConfig = { method: method, url: domain + url, data: param }
      }
      return axios(requestConfig)
        .then(function (res) {
          if (res.data.code !== 200) {
            var msg = res.data.msg || res.data.message || '请求失败'
            layer.msg(msg, { icon: 2 })
            throw new Error(msg)
          } else {
            return res.data.data
          }
        })
        .catch(function (err) {
          layer.msg(err.message, { icon: 2 })
          throw new Error(err)
        })
    }

            that.recommend_list = res.list;
          })
        },
        // 生成淘口令
        createTpwdCommon: function (item, type) {
          if (this.getParam('preview') == 1) {
            layer.msg('请生成推广海报', { icon: 2 });
            return;
          }
          var that = this;
          var id = this.id;
          var domain = window.location.host;
          var data = {
            id: id,
            domain: domain,
            type: type
          };
          if (type === 1) {
            data.itemid = item.itemid;
            data.activityid = item.activityid;
            data.title = item.itemshorttitle;
          } else if (type === 2) {
            data.activity_material_id = item.activity_material_id;
            data.activity_url = item.activity_url;
            data.title = item.activity_title;
          }
          this.API.GET_TPWD(data).then(function (res) {
            that.getTpwd(res.tpwd);
          });
        },
        // 获取淘口令
        getTpwd: function (str) {
          this.tpwd = str;
          this.copy_layer = layer.open({
            type: 1,
            closeBtn: 1,
            title: false,
            shadeClose: true,
            shade: 0.4,
            shift: 0,
            maxWidth: 999,
            content: $('.tpwd-copy-content')
          });
        },
        // 复制淘口令
        copyTpwd: function () {
          if (!this.tpwd) {
            return
          }
          var that = this
          var clipboard = new ClipboardJS('#public-copy', {
            text: function () {
              return that.tpwd
            }
          })
          clipboard.on('success', function (e) {
            layer.msg('复制成功，请打开淘宝APP领券购买', {
              time: 3000,
            }, function () {
              layer.close(that.copy_layer);
              that.tpwd = '';
            })
            e.clearSelection()
            clipboard.destroy() // 释放内存
          })
          clipboard.on('error', function (e) {
            layer.msg('复制失败，请手动复制', {
              shade: 0.4,
              time: 1000,
              shadeClose: true,
            }, function () {
              layer.open({
                type: 1,
                closeBtn: 1,
                title: false,
                shadeClose: true,
                shade: 0.4,
                shift: 0,
                maxWidth: 999,
                content: $('.tpwd-content')
              });
            })
            clipboard.destroy() // 释放内存
          })
        },
        // 跳转链接
        goLink: function (url) {
          if (this.getParam('preview') == 1) {
            return;
          }
          window.open(url)
        },
        // 跳转超级搜
        goSearchLink: function () {
          if (this.getParam('preview') == 1) {
            return;
          }
          var independent = "{$independent}";
          if (independent == 1) {
            var link = '/BenefitList/search?id=' + this.getParam('id');
          } else {
            var link = '/benefit_search.html?id=' + this.getParam('id');
          }
          var url = window.location.protocol.substring(0, window.location.protocol.length - 1) + '://' + window.location.host + link;
          window.open(url)
        },
        // 打开海报弹窗
        openPosterDialog: function () {
          if (this.getParam('preview') == 1) {
            return;
          }
          layer.open({
            type: 1,
            closeBtn: 1,
            title: false,
            shadeClose: true,
            shade: 0.4,
            shift: 0,
            maxWidth: 999,
            content: $('.poster-content')
          });
        }
      },
    })
